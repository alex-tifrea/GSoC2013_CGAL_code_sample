\ccCreationVariable{g}
\ccConstructor{Random_points_in_cube_d(int dim,  double a, Random& rnd =
  default_random);}{%
  $g$ is an input iterator creating points of type \ccc{Point_d} uniformly
  distributed in the half-open cube of dimension $dim$ with side length $2 a$, centered
  at the origin.
For every point $ p = \ccc{*g}$ and for all $ i<dim$ we have $ -a \le p[i] < a$. 
  $dim$ random numbers are needed from \ccc{rnd} for each point.
}
